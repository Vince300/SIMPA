215  public InputSequence getShortestCounterExemple(
216  State s1, Mealy a2, State s2) {
217  if (s1 == null)
218  s1 = currentState;
219  assert automata.isConnex() && a2.isConnex();
220  int maxLength = (automata.getStateCount() > a2.getStateCount() ? automata.getStateCount() : a2.getStateCount());
221  class Node{public InputSequence i; public State originalEnd; public State conjectureEnd;public String toString(){return "for input '" + i + "' this driver go to '" + originalEnd + "' and the other go to '"+conjectureEnd+"'\n";}}
222  LinkedList<Node> toCompute = new LinkedList<Node>();
223  Node n = new Node();
224  n.i = new InputSequence();
225  n.originalEnd = s1;
226  n.conjectureEnd = s2;
227  toCompute.add(n);
228  while (!toCompute.isEmpty()){
229  Node current = toCompute.pollFirst();
230  if (current.i.getLength() > maxLength)
231  continue;
232  for (String i : getInputSymbols()){
233  MealyTransition originalT = automata.getTransitionFromWithInput(current.originalEnd, i);
234  MealyTransition conjectureT = a2.getTransitionFromWithInput(current.conjectureEnd, i);
235  if (!originalT.getOutput().equals(conjectureT.getOutput())){
236  current.i.addInput(i);
237  return current.i;
238  }
239  Node newNode = new Node();
240  newNode.i = new InputSequence();
241  newNode.i.addInputSequence(current.i);
242  newNode.i.addInput(i);
243  newNode.originalEnd = originalT.getTo();
244  newNode.conjectureEnd = conjectureT.getTo();
245  toCompute.add(newNode);
246  }
247  }
248  return null;
