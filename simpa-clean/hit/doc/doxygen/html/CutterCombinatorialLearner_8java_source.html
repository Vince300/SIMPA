389  private InputSequence getShortestCounterExemple(Mealy a1,
390  State s1, Conjecture a2, State s2) {
391  assert a1.isConnex() && a2.isConnex();
392  int maxLength = (a1.getStateCount() > a2.getStateCount() ? a1.getStateCount() : a2.getStateCount());
393  class Node{public InputSequence i; public State originalEnd; public State conjectureEnd;}
394  LinkedList<Node> toCompute = new LinkedList<Node>();
395  Node n = new Node();
396  n.i = new InputSequence();
397  n.originalEnd = s1;
398  n.conjectureEnd = s2;
399  toCompute.add(n);
400  while (!toCompute.isEmpty()){
401  Node current = toCompute.pollFirst();
402  if (current.i.getLength() > maxLength)
403  continue;
404  for (String i : a2.getInputSymbols()){
405  MealyTransition originalT = a1.getTransitionFromWithInput(current.originalEnd, i);
406  MealyTransition conjectureT = a2.getTransitionFromWithInput(current.conjectureEnd, i);
407  if (!originalT.getOutput().equals(conjectureT.getOutput())){
408  current.i.addInput(i);
409  return current.i;
410  }
411  Node newNode = new Node();
412  newNode.i = new InputSequence();
413  newNode.i.addInputSequence(current.i);
414  newNode.i.addInput(i);
415  newNode.originalEnd = originalT.getTo();
416  newNode.conjectureEnd = conjectureT.getTo();
417  toCompute.add(newNode);
