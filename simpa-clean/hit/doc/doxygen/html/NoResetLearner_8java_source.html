295  public boolean checkExact(Mealy automata, State currentState){
296  LogManager.logStep(LogManager.STEPOTHER, "checking the computed conjecture is exactly equivalent");
297  class FoundState {
298  public State computedState; //a state in the conjecture
299  public List<String> uncheckedTransitions;
300  public FoundState(State s, List<String> I){
301  computedState = s;
302  uncheckedTransitions = new ArrayList<String>(I);
303  }
304  public String toString(){return computedState + " but transitions " + uncheckedTransitions + "have not been checked";}
305  }
306  //currentFoundState is maintained in order that currentFoundState.computedState is the current state in conjecture
307  FoundState currentFoundState = new FoundState(dataManager.getC(dataManager.traceSize()).getState(),driver.getInputSymbols());
308  //assigned is a table to associate a FoundState to each state in the given automata
309  Map<State,FoundState> assigned = new HashMap<State,FoundState>();
310  assigned.put(currentState, currentFoundState);
311  State uncheckedState = currentState; //a state with an unchecked transition
312  List<String> path = new ArrayList<String>();//the path from the current state to uncheckeState
313 
314  //now we iterate over all unchecked transitions
315  while (uncheckedState != null) {
316  FoundState uncheckedFoundState = assigned.get(uncheckedState);
317  LogManager.logInfo("Applying " + path + "in order to go in state " + uncheckedState + " and then try " + uncheckedFoundState.uncheckedTransitions.get(0));
318  path.add(uncheckedFoundState.uncheckedTransitions.get(0));
319 
320  //we follow path in driver (the conjecture) and the given automata
321  for (String i : path){
322  currentFoundState.uncheckedTransitions.remove(i);
323  MealyTransition t = automata.getTransitionFromWithInput(currentState, i);
324  currentState = t.getTo();
325  String o = dataManager.apply(i);
326  if (! o.equals(t.getOutput())){
327  LogManager.logInfo("expected output was " + t.getOutput());
328  return false;
329  }
330  currentFoundState = assigned.get(currentState);
331  if (currentFoundState == null){
332  currentFoundState = new FoundState(dataManager.getC(dataManager.traceSize()).getState(), driver.getInputSymbols());
333  assigned.put(currentState, currentFoundState);
334  } else if (currentFoundState.computedState != dataManager.getC(dataManager.traceSize()).getState()){
335  LogManager.logInfo("it was expected to arrive in " + t.getTo());
336  return false;
337  }
338  }
339  //now we've applied an unchecked transition (which is now checked)
340 
341  //and then we compute a new path to go to another state with unchecked transitions
342  class Node {public List<String> path; public State state;}
343  LinkedList<Node> nodes = new LinkedList<Node>();
344  Node node = new Node();
345  node.path = new ArrayList<String>();
346  node.state = currentState;
347  nodes.add(node);
348 
349  Map<State, Boolean> crossed = new HashMap<State, Boolean>();//this map is used to store the node crossed during the path searching (avoid going to the same state by two different path)
350  for (State s : automata.getStates())
351  crossed.put(s,false);
352  uncheckedState = null;
353  path = null;
354  while (!nodes.isEmpty()){
355  node = nodes.pollFirst();
356  if (!assigned.get(node.state).uncheckedTransitions.isEmpty()){
357  uncheckedState = node.state;
358  path = node.path;
359  break;
360  }
361  if (crossed.get(node.state))
362  continue;
363  for (String i : driver.getInputSymbols()){
364  Node newNode = new Node();
365  newNode.path = new ArrayList<String>(node.path);
366  newNode.path.add(i);
367  newNode.state = automata.getTransitionFromWithInput(node.state, i).getTo();
368  nodes.add(newNode);
369  crossed.put(node.state, true);
370  }
371  }
372  }
373 
374  return true;
375  }
