42  private Map<State, Map<State, State>> createStates(){
43  Map<State,Map<State,State>> combinedStates = new HashMap<State,Map<State,State>>();
44  for (State s1 : m1.getStates()){
45  Map<State,State> s1States = new HashMap<State,State>();

49  s1States.put(s2, s1s2State);

51  }

57  private void createTransitions(Map<State,Map<State,State>> combinedStates){
58  for (int i = 0; i < 2; i++){
59  Mealy m_a, m_b;
60  if (i == 0){
61  m_a = m1;
62  m_b = m2;
63  }else{
64  m_a = m2;
65  m_b = m1;
66  }
67  for (MealyTransition t_a : m_a.getTransitions()){
68  for (State s_b : m_b.getStates()){
69  State from;
70  if (i == 0)
71  from = combinedStates.get(t_a.getFrom()).get(s_b);
72  else
73  from = combinedStates.get(s_b).get(t_a.getFrom());
74  State to_b;
75  String output = t_a.getOutput();
76  MealyTransition t_b = m_b.getTransitionFromWithInput(s_b, t_a.getInput());
77  if (t_b == null){ // we are making an asynchronous product
78  to_b = s_b;
79  }else{
80  to_b = t_b.getTo();
81  output += "_x_" + t_b.getOutput();
82  }
83  State to;
84  if (i == 0)
85  to = combinedStates.get(t_a.getTo()).get(to_b);
86  else
87  to = combinedStates.get(to_b).get(t_a.getTo());
88  MealyTransition t = new MealyTransition(this, from, to, t_a.getInput(),output);
89  if (getTransitionFromWithInput(from, t_a.getInput()) == null)//because in case of synchronous product we will add to times the transition with reversed output
90  addTransition(t);
91  }
92  }
