249  public WebRequest parameterizedInputToRequest(ParameterizedInput pi) {

251  WebInput in = inputsFromSymbols.get(pi.getInputSymbol());
252  HttpMethod method = in.getMethod();
253  WebRequest request = null;

257  pi = pi.clone();

326  .item(0).getTextContent());
327  config.setLimitSelector(root.getElementsByTagName("limitSelector")
328  .item(0).getTextContent());
329  config.setCookies(root.getElementsByTagName("cookies").item(0)
330  .getTextContent());
331 
332  Node inputsNode = root.getElementsByTagName("inputs").item(0);
333  NodeList inputNodesList = inputsNode.getChildNodes();
334  for (int i = 0; i < inputNodesList.getLength(); i++) {
335  if (!inputNodesList.item(i).getNodeName().equals("input")) {
336  continue;
337  }
338 
339  Node inputNode = inputNodesList.item(i);
340  WebInput in = new WebInput();
341  in.setAddress(inputNode.getAttributes()
342  .getNamedItem("address").getNodeValue());
343  in.setMethod(HttpMethod.valueOf(inputNode
344  .getAttributes().getNamedItem("method")
345  .getNodeValue()));
346  in.setType(Type.valueOf(inputNode.getAttributes()
347  .getNamedItem("type").getNodeValue()));
348 
349  int nbValue = 0;
350  if (!inputNode.getChildNodes().item(1).getNodeName().equals("parameters")) {
351  continue;
352  }
353 
354  Node parametersNode = inputNode.getChildNodes().item(1);
355  NodeList parametersCombinationNodesList = parametersNode.getChildNodes();
356  for (int j = 0; j < parametersCombinationNodesList.getLength(); j++) {
357  if (!parametersCombinationNodesList.item(j).getNodeName().equals("parametersCombination")) {
358  continue;
359  }
360  Node parametersCombinationNode = parametersCombinationNodesList.item(j);
361  nbValue++;
362 
363  for (int k = 0; k < parametersCombinationNode.getChildNodes().getLength(); k++) {
364  if (!parametersCombinationNode.getChildNodes().item(k).getNodeName().equals("parameter")) {
365  continue;
366  }
367 
368  Node parameterNode = parametersCombinationNode.getChildNodes().item(k);
369  String name = parameterNode.getAttributes().getNamedItem("name").getNodeValue();
370  String value = parameterNode.getTextContent();
371  if (in.getParams().get(name) == null) {
372  in.getParams().put(name,
373  new ArrayList<String>());
374  }
375  in.getParams().get(name).add(value);
376  }
377  }
378  in.setNbValues(nbValue);
379  if (in.getParams().isEmpty()) {
380  in.getParams().put(NO_PARAM_NAME,
381  Utils.createArrayList(NO_PARAM_VALUE));
382  in.setNbValues(1);
383  }
384  this.inputs.add(in);
385  }
386 
387  NodeList outputNodesList = root.getElementsByTagName("outputs").item(0)
388  .getChildNodes();
389  for (int i = 0; i < outputNodesList.getLength(); i++) {
390  Node outputNode = outputNodesList.item(i);
391  if (!outputNode.getNodeName().equals("output")) {
392  continue;
393  }
394  String source = outputNode.getChildNodes().item(1).getTextContent();
395  WebOutput out = new WebOutput(source);
396  NodeList parameterNodesList = outputNode.getChildNodes().item(3).getChildNodes();
397  for (int j = 0; j < parameterNodesList.getLength(); j++) {
398  if (!parameterNodesList.item(j).getNodeName().equals("parameter")) {
399  continue;
400  }
401  String value = parameterNodesList.item(j).getTextContent();
402  out.addParam(value);
403  }
404  this.outputs.add(out);
405 
406  }
407  } catch (ParserConfigurationException | SAXException e) {
408  LogManager.logException("Error parsing the xml file \"" + xml + "\"", e);
409  } catch (IOException e) {
410  LogManager.logException("Unable to read the file \"" + xml + "\"", e);
411  }
412  return config;

477  private void generateSymbols() {
478  inputSymbols = new HashMap<>();
479  for (WebInput i : inputs) {
480  String[] splitedAddress = i.getAddress().split("/");
481  String pageName = splitedAddress[splitedAddress.length - 1];
482  //To be compatible with ASLan++
483  pageName = pageName.replaceAll("[^a-zA-Z_0-9]", "_");
484  if (inputSymbols.containsValue(pageName)) {
485  int counter = 2;
486  while (inputSymbols.containsValue(pageName + "_" + counter)) {
487  counter++;
488  }
489  inputSymbols.put(i, pageName + "_" + counter);
490  } else {
491  inputSymbols.put(i, pageName);
492  }
493  }
494 
495  outputSymbols = new HashMap<>();
496  int i = 0;
497  for (WebOutput o : outputs) {
498  outputSymbols.put(o, "output_" + i);
499  i++;
500  }
501 
502 
503  inputsFromSymbols = new HashMap<>();
504  for (Map.Entry<WebInput, String> entry : inputSymbols.entrySet()) {
505  WebInput key = entry.getKey();
506  String value = entry.getValue();
507  if (inputsFromSymbols.put(value, key) != null) {
508  throw new AssertionError("The symbol " + value + " should be unique");
509  }
510  }
511  }
