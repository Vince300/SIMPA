388  private static byte[] encode3to4(byte[] source, int srcOffset,
389  int numSigBytes, byte[] destination, int destOffset, int options) {
390 
391  byte[] ALPHABET = getAlphabet(options);
392 
393  // 1 2 3
394  // 01234567890123456789012345678901 Bit position
395  // --------000000001111111122222222 Array position from threeBytes
396  // --------| || || || | Six bit groups to index ALPHABET
397  // >>18 >>12 >> 6 >> 0 Right shift necessary
398  // 0x3f 0x3f 0x3f Additional AND
399 
400  // Create buffer with zero-padding if there are only one or two
401  // significant bytes passed in the array.
402  // We have to shift left 24 in order to flush out the 1's that appear
403  // when Java treats a value as negative that is cast from a byte to an
404  // int.
405  int inBuff = (numSigBytes > 0 ? ((source[srcOffset] << 24) >>> 8) : 0)
406  | (numSigBytes > 1 ? ((source[srcOffset + 1] << 24) >>> 16) : 0)
407  | (numSigBytes > 2 ? ((source[srcOffset + 2] << 24) >>> 24) : 0);
408 
409  switch (numSigBytes) {
410  case 3:
411  destination[destOffset] = ALPHABET[(inBuff >>> 18)];
412  destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
413  destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
414  destination[destOffset + 3] = ALPHABET[(inBuff) & 0x3f];
415  return destination;
416 
417  case 2:
418  destination[destOffset] = ALPHABET[(inBuff >>> 18)];
419  destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
420  destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
421  destination[destOffset + 3] = EQUALS_SIGN;
422  return destination;
423 
424  case 1:
425  destination[destOffset] = ALPHABET[(inBuff >>> 18)];
426  destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
427  destination[destOffset + 2] = EQUALS_SIGN;
428  destination[destOffset + 3] = EQUALS_SIGN;
429  return destination;
430 
431  default:
432  return destination;
433  } // end switch
434  } // end encode3to4

551  public static String encodeObject(java.io.Serializable serializableObject,
552  int options) throws java.io.IOException {
553 
554  if (serializableObject == null) {
555  throw new NullPointerException("Cannot serialize a null object.");
556  } // end if: null
557 
558  // Streams
559  java.io.ByteArrayOutputStream baos = null;
560  java.io.OutputStream b64os = null;
561  java.util.zip.GZIPOutputStream gzos = null;
562  java.io.ObjectOutputStream oos = null;
563 
564  try {
565  // ObjectOutputStream -> (GZIP) -> Base64 -> ByteArrayOutputStream
566  baos = new java.io.ByteArrayOutputStream();
567  b64os = new Base64.OutputStream(baos, ENCODE | options);
568  if ((options & GZIP) != 0) {
569  // Gzip
570  gzos = new java.util.zip.GZIPOutputStream(b64os);
571  oos = new java.io.ObjectOutputStream(gzos);
572  } else {
573  // Not gzipped
574  oos = new java.io.ObjectOutputStream(b64os);
575  }
576  oos.writeObject(serializableObject);
577  } // end try
578  catch (java.io.IOException e) {
579  // Catch it and then throw it immediately so that
580  // the finally{} block is called for cleanup.
581  throw e;
582  } // end catch
583  finally {
584  try {
585  oos.close();
586  } catch (Exception e) {
587  }
588  try {
589  gzos.close();
590  } catch (Exception e) {
591  }
592  try {
593  b64os.close();
594  } catch (Exception e) {
595  }
596  try {
597  baos.close();
598  } catch (Exception e) {
599  }
600  } // end finally
601 
602  // Return value according to relevant encoding.
603  try {
604  return new String(baos.toByteArray(), PREFERRED_ENCODING);
605  } // end try
606  catch (java.io.UnsupportedEncodingException uue) {
607  // Fall back to some Java default
608  return new String(baos.toByteArray());
609  } // end catch
610 
611  } // end encode
