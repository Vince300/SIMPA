446  public InputSequence getShortestAlpha(FullyQualifiedState s) {
447  assert s != null;
448  class Node{
449  public InputSequence path;
450  public FullyQualifiedState end;
451  public boolean equals(Object o){ if (o instanceof Node) return equals((Node) o); return false;}
452  public boolean equals(Node o){return path.equals(o.path) && end.equals(o.end);}

455  class PathComparator implements Comparator<Node>{
456  @Override
457  public int compare(Node o1, Node o2) {
458  int diff = o1.path.getLength() - o2.path.getLength();
459  return diff;
460  }
461  }
462  PriorityQueue<Node> paths=new PriorityQueue<Node>(10, new PathComparator());
463  Node firstNode = new Node();
464  firstNode.end = s;
465  firstNode.path = new InputSequence();
466  paths.add(firstNode);
467  List<FullyQualifiedState> reachedStates = new ArrayList<FullyQualifiedState>();
468  while(!paths.isEmpty()){
469  firstNode = paths.poll();
470  if (reachedStates.contains(firstNode.end))
471  continue;
472  reachedStates.add(firstNode.end);
473  if (!firstNode.end.getUnknowTransitions().isEmpty()){
474  LogManager.logInfo("chosen alpha is " + firstNode.path + " that lead in " + firstNode.end);
475  return firstNode.path;
476  }
477  for (FullyKnownTrace t : firstNode.end.getVerifiedTrace()){
478  Node childNode = new Node();
479  childNode.end = t.getEnd();

504  if (DataManager.instance.getStates().size() != DataManager.instance.maxStates){
505  LogManager.logInfo("cannot deduce while there is unknown states");

510  for (String i : sn.getKnownInputs()){
511  AbstractNode child = sn.getChild(i);
512  if (child.isStateNode())
513  continue;
514  List<StateNode> incompatibleNodes = child.getIncompatibleStateNode();
515  if (incompatibleNodes.size() + 1 == DataManager.instance.maxStates){
516  LogManager.logInfo("Found a node which is almost incompatible with all others ! "+sn+" followed by "+i+"/"+sn.getOutput(i)+" give "+child+" which is incompatible with "+incompatibleNodes);
517  FullyQualifiedState end = null;

519  if (!incompatibleNodes.contains(fqs.getVNode()))
520  end = fqs;
