158  LogManager.logConsole("Building the raw conjecture");
159  LiConjecture c = new LiConjecture(driver);
160  for (int i = 0; i < cTable.getCountOfRowsInS(); i++) {
161  c.addState(new State("S" + i, cTable.getRowInS(i).isEpsilon()));
162  }
163  List<LiControlTableRow> allRows = cTable.getAllRows();
164  Collections.sort(allRows, new Comparator<LiControlTableRow>() {
165  @Override
166  public int compare(LiControlTableRow o1, LiControlTableRow o2) {
167  return o1.getPIS().sequence.size()
168  - o2.getPIS().sequence.size();
169  }
170  });
171  for (LiControlTableRow ctr : allRows) {
172  if (ctr.isEpsilon()) {
173  continue;
174  }
175  int iFrom = cTable.getFromState(ctr);
176  int iTo = cTable.getToState(ctr);
177  State from = c.getState(iFrom);
178  State to = c.getState(iTo);
179  String inputSymbol = ctr.getLastPI().getInputSymbol();
180  ArrayList<LiControlTableItem> allControlItems = cTable
181  .getRowInS(iFrom).getColumn(
182  cTable.getInputSymbols().indexOf(inputSymbol));
183  ArrayList<LiDataTableItem> allDataItems = dTable.getRowInS(
184  iFrom).getColumn(
185  cTable.getInputSymbols().indexOf(inputSymbol));
186  for (int i = 0; i < cTable.R.size(); i++) {
187  if (cTable.getRowInS(iFrom).isEquivalentTo(
188  cTable.getRowInR(i))) {
189  allDataItems.addAll(dTable.getRowInR(i).getColumn(
190  cTable.getInputSymbols().indexOf(inputSymbol)));
191  }
192  }
193 
194  for (LiControlTableItem cti : allControlItems) {
195  if (cti.isOmegaSymbol()) {
196  continue;
197  }
198  if (!ctr.getLastPI().getParameters()
199  .equals(cti.getParameters())) {
200  continue;
201  }
202  ArrayList<LiDataTableItem> correspondingDataTableItems = new ArrayList<>();
203  for (LiDataTableItem dti : allDataItems) {
204  if (cti.getOutputSymbol().equals(
205  dti.getOutputSymbol())
206  && !correspondingDataTableItems
207  .contains(dti)) {
208  correspondingDataTableItems.add(dti);
209  }
210  }
211  for (NDV n : dTable.ndvList) {
212  ParameterizedInputSequence tmpis = n.getPIS();
213  if (ctr.getPIS()
214  .toString()
215  .equals(tmpis.removeEmptyInput()
216  .toString())) {
217  for (LiDataTableItem tmp : correspondingDataTableItems) {
218  tmp.getOutputParameters().set(
219  n.paramIndex,
220  new Parameter("Ndv"
221  + n.indexNdv,
222  Types.STRING));
223  }
224  }
225  }
226  if (!correspondingDataTableItems.isEmpty()) {
227  c.addTransition(new EFSMTransition(c, from, to,
228  inputSymbol, cti.getOutputSymbol(),
229  correspondingDataTableItems));
230  }
231  }
232  }
233  for (int i = c.getTransitionCount() - 1; i >= 0; i--) {
234  for (int j = i - 1; j >= 0; j--) {
235  EFSMTransition t1 = c.getTransition(i);
236  EFSMTransition t2 = c.getTransition(j);
237  if (t1.getInput().equals(t2.getInput())
238  && t1.getOutput().equals(t2.getOutput())
239  && (t1.getFrom().equals(t2.getFrom()))
240  && (t1.getTo().equals(t2.getTo()))) {
241  c.removeTransition(i);
242  break;
243  }
244  }
245  }
246 
247  expandInitParams(c.getTransitions());
248 
249  LogManager.logInfo("Raw conjecture has " + c.getStateCount()
250  + " states and " + c.getTransitionCount() + " transitions : ");
251  for (EFSMTransition t : c.getTransitions()) {
252  LogManager.logTransition(t.toString());
253  for (LiDataTableItem dti : t.getParamsData()) {
254  LogManager.logData(dti.toString());
255  }
256  }
257  LogManager.logLine();
258 
259  c.exportToRawDot();
260  c.exportToDot();
261  c.exportToAslan();
262  c.exportToXML();
263  return c;

371  private void processQuery(ParameterizedInputSequence query) {
372  TreeMap<String, List<Parameter>> automataState = driver.getInitState();
373  //Sends the sequence and store the results in a POS
374  //Before sending, search a velue to give to each NDV
375  boolean needToReset = true;
376  ParameterizedInputSequence pis = new ParameterizedInputSequence();
377  ParameterizedOutputSequence pos = new ParameterizedOutputSequence();
378  List<ParameterizedInput> notSent = new LinkedList<>();
379  for (ParameterizedInput queryPI : query.sequence) {
380  ParameterizedInput api = queryPI.clone();
381  for (int k = 0; k < api.getParameters().size(); k++) {
382  if (api.isNdv(k)) {
383  //System.out.println("Requesting NDV for " + api.getInputSymbol());
384  NDV correspondingNDV = dTable.ndvList.get(api.getNdvIndexForVar(k));
385  /* Looks in the current automata state the last value
386  of the parameter returned by the output symbol in which the NDV
387  was found */
388  Parameter ndvValue = automataState.get(correspondingNDV.getOutputSymbol()).get(correspondingNDV.paramIndex);
389  /* If this symbol was not encountered before, "init" is present in the
390  automata, and there is no need to send this dummy value
391  to the real system. So we replace it by a default value.*/
392  if (ndvValue.isInit()) {
393  ndvValue = defaultParamValues.get(api.getInputSymbol()).get(0).get(k).clone();
394  }
395  ndvValue.setNdv(correspondingNDV.indexNdv);
396  api.setParameterValue(k, ndvValue);
397  }
398  }
399  pis.addParameterizedInput(api);
400 
401  /* Checks if the formed input sequence has already been sent */
402  LiControlTableItem cti;
403  ParameterizedOutput po;
404  if (LESS_REQUESTS && (cti = cTable.getItem(pis)) != null) {
405  String outputSymbol = cti.getOutputSymbol();
406  List<Parameter> outputParams = dTable.getOutputParametersFor(pis, automataState);
407  if (outputParams == null) {
408  LogManager.logFatalError("output params not found for "+pis+" "+automataState);
409  throw new AssertionError("output params not found for "+pis+" "+automataState);
410  }
411  po = new ParameterizedOutput(outputSymbol, outputParams);
412  notSent.add(api);
413  } else {
414  if (needToReset) {
415  driver.reset();
416  needToReset = false;
417  }
418  for (ParameterizedInput pi : notSent) {
419  driver.execute(pi);
420  //TODO : complete the data table with real values
421  }
422 
423  po = driver.execute(api);
424  }
425  pos.addParameterizedOuput(po);
426 
427  //Store the state in which the automata was before executing the PI
428  LiDataTableItem dti = new LiDataTableItem(
429  pis.getLastParameters(),
430  (TreeMap<String, List<Parameter>>) automataState.clone(),
431  pos.getLastParameters(),
432  pos.getLastSymbol());
433  updateDataTable(dti, pis);
434 
435  //Update the automata
436  automataState.put(pis.getLastSymbol(), pis.getLastParameters());
437  if (!pos.sequence.get(pos.sequence.size() - 1).isOmegaSymbol()) {
438  automataState.put(pos.getLastSymbol(), pos.getLastParameters());
439  }
440  }
441 
442  //Completes the control table with the new item
443  LiControlTableItem cti = new LiControlTableItem(
444  pis.getLastParameters(), pos.getLastSymbol());//cf TODO(*)
445  cTable.addAtCorrespondingPlace(cti, pis);
446  }

564  }

605  //bugfix : this condition prevents adding to S multiple non-closed rows
606  //representing the same state in the same instance of this loop
607  //TODO : simplify this loop ? (with cTable.getFirstNonClosedRow for example)
608  LogManager.logStep(LogManager.STEPNCR,
609  cTable.R.get(nonClosedRowRealIndex).getPIS());
610  handleNonClosed(nonClosedRowRealIndex);

612  } else {

619  }

629  }
630  }

632  if (xssDetector.detectReflections()) {
633  LogManager.logInfo("[XSS] Trying to confirm reflections");
634  xssDetector.confirmReflections();

636  xssDetector.testReflections();

766  private int seemsEquivalent(int nonClosedRowIndex) {
767  List<Double> stats = new ArrayList<Double>();
768  LiControlTableRow nonClosedRow = cTable.getRowInR(nonClosedRowIndex);
769  double max = 0;
770  //for each row of S ...
771  for(LiControlTableRow rowOfS : cTable.S){
772  //for (int i = 0; i < cTable.getCountOfRowsInS(); i++) {
773  double diff = 0.0;
774  max = 0;
775  //for each column of the table ...
776  for (int j = 0; j < cTable.getColsCount(); j++) {
777  //... we count a difference for each output symbol in the j^th
778  //cell of the non-closed row that is not in the j^th cell of the
779  //row of S.
780  for (int k = 0; k < nonClosedRow.getSizeOfColumn(j); k++) {
781  max++;
782  if (rowOfS.getSizeOfColumn(j) <= k){
783  diff++;
784  } else {
785  String outputSymbolRowInS = rowOfS.getColumn(j).get(k).getOutputSymbol();
786  String outputSymbolNonClosedRow = nonClosedRow.getColumn(j).get(k).getOutputSymbol();
787  if (!outputSymbolRowInS.equals(outputSymbolNonClosedRow)) {
788  diff++;
789  }
790  }
791  }
792  }
793  //store the "difference ratio" between the non-closed row and the row of S
794  stats.add(diff / max);
795  }
796 
797  //select the index of the row of S that is the closest to the non-closed row
798  int indexMin = 0;
799  for (int i = 1; i < stats.size(); i++) {
800  if (stats.get(i) < stats.get(indexMin)) {
801  indexMin = i;
802  }
803  }
804 
805  return (stats.get(indexMin) <= 1 / max ? indexMin : -1);
806  }
