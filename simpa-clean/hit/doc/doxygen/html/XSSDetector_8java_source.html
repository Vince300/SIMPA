99  public boolean detectReflections() {
100  boolean reflectionsHaveBeenFound = false;
101  /* Iterate on the new items */
102  while (!itemsToCheck.isEmpty()) {
103  SimplifiedDataItem currentItem = itemsToCheck.pop();
104  List<Parameter> outputParameters = currentItem.result.getParameters();
105  for (Parameter outputParameter : outputParameters) {
106  String outputValue = outputParameter.value;
107  if (ignoredValues.contains(outputValue)) {
108  continue;
109  }
110 
111  /* Iterate on the sequence of inputs */
112  List<ParameterizedInput> path = currentItem.path.clone().sequence;
113  for (ParameterizedInput pi : path) {
114  List<Parameter> piParameters = pi.getParameters();
115 
116  /* Iterate on the different parameter of the input */
117  for (Parameter param : piParameters) {
118  String inputValue = param.value;
119 
120  if (inputValue.length() < MINIMAL_SIZE && ignoredValues.contains(inputValue)) {
121  continue;
122  }
123 
124  if (isReflected(inputValue, outputValue)) {
125  Reflection newReflection
126  = new Reflection(currentItem.path.clone().sequence,
127  path.indexOf(pi),
128  piParameters.indexOf(param),
129  outputParameters.indexOf(outputParameter),
130  currentItem.result.getOutputSymbol());
131 
132  boolean foundSimilar = false;
133  for (Reflection reflection : potentialReflectionsFound) {
134  if (newReflection.path.containsAll(reflection.path)
135  && newReflection.path.get(newReflection.inputElementIndex).getInputSymbol()
136  .equals(
137  reflection.path.get(reflection.inputElementIndex).getInputSymbol())
138  && newReflection.inputElementParamIndex == reflection.inputElementParamIndex
139  && newReflection.expectedOutputSymbol.equals(reflection.expectedOutputSymbol)
140  && reflection.outputElementParamIndex == newReflection.outputElementParamIndex) {
141  foundSimilar = true;
142  break;
143  }
144  }
145 
146  if (!foundSimilar && !potentialReflectionsFound.contains(newReflection)) {
147  LogManager.logInfo("[XSS] Potential reflection found : ");
148  LogManager.logInfo("[XSS]\t" + (newReflection.inputElementParamIndex + 1) + "th parameter of input "
149  + currentItem.path.sequence.get(newReflection.inputElementIndex));
150  LogManager.logInfo("[XSS]\tis reflected in the " + (newReflection.outputElementParamIndex + 1) + "th parameter of the output "
151  + currentItem.result);
152  LogManager.logInfo("[XSS]\t" + "in the sequence :");
153  LogManager.logInfo(currentItem.path.toString());
154 
155  potentialReflectionsFound.add(newReflection);
156  reflectionsHaveBeenFound = true;
157  }
158  }
159  }
160  }
161  }
162  }
163  return reflectionsHaveBeenFound;

175 
176  //retrieve the input parameter that is to be reflected
177  List<ParameterizedInput> sequence = potentialReflection.path;
178  Parameter inputParameter = sequence.get(potentialReflection.inputElementIndex)
179  .getParameters().get(potentialReflection.inputElementParamIndex);
180 
181  //replace its value by the randomly generated one
182  inputParameter.value = randomString;
183 
184  //test if this value is still reflected
185  driver.reset();
186  ParameterizedOutput result = null;
187  boolean abort = false;

265  StringBuilder completePayload = new StringBuilder(startPattern);

269  String response = "";

271  for (int i = 0; i < reflection.path.size(); i++) {
272  ParameterizedInput pi = reflection.path.get(i);

274  pi.getParameters().get(reflection.inputElementParamIndex).value = completePayload.toString();
275  }
276  response = driver.submit(pi, true);
277  }

279  LogManager.logInfo("[XSS] Payload \'" + xssPayload + "\' reflected as \'" + expectedReflection + "\', as exepected");
280  LogManager.logInfo("[XSS] SUCCESS : XSS found for reflection " + reflection);
281  LogManager.logFoundXSS(reflection, driver);

283 
284  /* Both patterns were found : the data in between should contain our payload */
285  if (response.contains(startPattern) && response.contains(endPattern)) {
286  int indexStartFilteredPayload = response.indexOf(startPattern) + startPattern.length();
287  int indexEndFilteredPayload = response.indexOf(endPattern);
288  String filteredPayload = response.substring(indexStartFilteredPayload, indexEndFilteredPayload); //TODO : handle multiple reflexions

291 
292  /* Only one pattern was found, our payload should be located before or after */
293  } else if (response.contains(startPattern) ^ response.contains(endPattern)) {

297  stringAroundPattern = response.substring(
298  indexStartFilteredPayload,

306  indexEndFilteredPayload);
307  }

309  //TODO : find behaviour

311  } else {
