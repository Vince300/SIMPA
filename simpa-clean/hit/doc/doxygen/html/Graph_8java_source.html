226  private StringBuilder makeDataDescritption(StatsSet s, List<Attribute<?>> ignoreFields) {
227  if (s.size() == 0) {
228  return new StringBuilder("No Data");
229  }
230  StringBuilder r = new StringBuilder();
231  String separator = "\\n";
232  List<Attribute<?>> fields = (dataDescriptionfields == null) ? Arrays.asList(s.getStats().get(0).getAttributes())
233  : dataDescriptionfields;
234  for (Attribute<?> a : fields) {
235  if (ignoreFields.contains(a) && dataDescriptionfields == null)
236  continue;
237  if ((!a.isParameter() || a.isVirtualParameter()) && dataDescriptionfields == null)
238  continue;
239  Comparable<?> min = s.attributeMin(a);
240  Comparable<?> max = s.attributeMax(a);
241  if (min.equals(max)) {
242  r.append(a.getName() + " : " + min + " " + a.getUnits().getSymbol() + separator);
243  } else {
244  r.append(min.toString() + " $\\\\\\\\leq$ " + a.getName() + " $\\\\\\\\leq$ " + max.toString() + " "
245  + escapeTex(a.getUnits().getSymbol()) + separator);
246  }
247  }
248  return r;
249  }

251  private File makeDataFile(StatsSet stats, PlotStyle style) {
252  File tempPlot;
253  PrintWriter tempWriter;
254  try {
255  tempPlot = File.createTempFile("simpa_" + ord + "_" + abs + "_", ".dat");
256  tempPlot.deleteOnExit();
257  toDelete.add(tempPlot);
258  tempWriter = new PrintWriter(tempPlot, "UTF-8");
259  } catch (IOException ioe) {
260  LogManager.logException("unable to create temporary file for gnuplot", ioe);
261  return null;
262  }
263  switch (style) {
264  case POINTS:
265  for (StatsEntry s : stats.getStats()) {
266  tempWriter.write(s.get(abs) + " " + s.get(ord) + "\n");
267  }
268  break;
269  case AVERAGE: {
270  Map<T_ABS, StatsSet> sorted = stats.sortByAtribute(abs);
271  List<T_ABS> keys = new ArrayList<T_ABS>(sorted.keySet());
272  Collections.sort(keys);
273  for (T_ABS key : keys) {
274  tempWriter.write(key + " " + sorted.get(key).attributeAVG(ord) + "\n");
275  }
276 
277  }
278  break;
279  case AVERAGE_WITH_EXTREMA: {
280  Map<T_ABS, StatsSet> sorted = stats.sortByAtribute(abs);
281  List<T_ABS> keys = new ArrayList<T_ABS>(sorted.keySet());
282  Collections.sort(keys);
283  for (T_ABS key : keys) {
284  StatsSet entrie = sorted.get(key);
285  tempWriter.write(key + " " + entrie.attributeAVG(ord) + " " + entrie.attributeMin(ord) + " "
286  + entrie.attributeMax(ord) + "\n");
287  }
288  }
289  break;
290  case MEDIAN: {
291  Map<T_ABS, StatsSet> sorted = stats.sortByAtribute(abs);
292  List<T_ABS> keys = new ArrayList<T_ABS>(sorted.keySet());
293  Collections.sort(keys);
294  for (T_ABS key : keys) {
295  tempWriter.write(key + " " + sorted.get(key).attributeMedian(ord) + "\n");
296  }
297  }
298  break;
299  case SMOOTH: {
300  Map<T_ABS, StatsSet> sorted = stats.sortByAtribute(abs);
301  List<T_ABS> keys = new ArrayList<T_ABS>(sorted.keySet());
302  Collections.sort(keys);
303  int n = 0;
304  String min = "";
305  StatsSet tmpSet = new StatsSet();
306  for (T_ABS key : keys) {
307  for (StatsEntry s : sorted.get(key).getStats()) {
308  if (n == 0) {
309  min = s.get(abs).toString();
310  tmpSet = new StatsSet();
311  }
312  tmpSet.add(s);
313  n++;
314  if (n == 5) {
315  float avg = tmpSet.attributeAVG(ord);
316  String max = s.get(abs).toString();
317  tempWriter.write(min + " " + avg + "\n");
318  tempWriter.write(max + " " + avg + "\n");
319  n = 0;
320  }
321  }
322  }
323  }
324  break;
325  default:
326  break;
327  }
328 
329  tempWriter.close();
330  return tempPlot;
331  }
