67 
68  //prepares the chain of input to send

70  WebInput curr = input;
71  if (withoutLast) {
72  curr = curr.getPrev();
73  }
74  while (curr != null) {
75  inputChain.addFirst(curr);
76  curr = curr.getPrev();
77  }
78 
79  String resultString = null;
80  for (WebInput currentInput : inputChain) {
81  checkInputParameters(currentInput);
82  resultString = sendInput(currentInput);
83  if (resultString == null || resultString.equals("")) {
84  return null;
85  }
86  /* If it is possible to reset the application, then sending the same
87  sequence of input should result in the same page with the same output parameters.
88  Therefore, if any parameter is different than the last time the same
89  input sequence was sent, it is a non-deterministic value and can be removed */
90  if (NDV_ANALYSIS
91  && config.getReset() != null
92  && currentInput != input
93  && currentInput.getOutput() != null
94  && currentInput.getOutput().getParamsNumber() > 0) {
95  WebOutput currentOutput = new WebOutput(resultString, config.getLimitSelector());
96  WebOutput storedOutput = currentInput.getOutput();
97  for (Iterator<String> iter = storedOutput.getParamsIterator(); iter.hasNext();) {
98  String paramPath = iter.next();
99  if (currentOutput.extractParam(paramPath) == null) {
100  continue;
101  }
102  if (!currentOutput.extractParam(paramPath).equals(storedOutput.getParamValue(paramPath))
103  && !Utils.isCSRFtoken(currentOutput.extractParam(paramPath))) {
104  //System.err.println("The param \"" + storedOutput.getParamValue(paramPath) + "\" has been detected as an ndv and is now removed");
105  storedOutput.removeParam(paramPath, iter);
106  }
107  }
108  }
109  }
110  if(withoutLast) {
111  return null;
112  } else {
113  return new WebOutput(resultString, input, config.getLimitSelector());
